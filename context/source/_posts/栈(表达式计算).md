### 栈
栈的思想: 先进先出

### 算法
将一个中缀表达式转换成后缀表达式, 然后求出其和.

### 基本思想
首先写一个类用来将中缀编程后缀表达式，涉及到的函数主要包含： 
    - 出栈 
    - 入栈 
    - 获得栈顶元素  
    - 栈内栈外元素的优先级， 栈内的元素优先级是增加， 栈外的元素优先级是减少 标志位#是最低优先级
    - 中缀变后缀的实现函数
        - 从表达式获得一个字符，如果数字直接进入后缀表达式
        - 如果栈内元素 小于 栈外元素，则栈外元素入栈
        - 如果栈顶元素 大于 栈外元素，则栈内元素出栈, 符号直接写到后缀表达式， 重复此操作符
        - 如果栈顶元素 等于 栈外元素, 则出栈即可.

再写一个计算和的类
    - 如果是数字直接进栈
    - 如果是符号，则直接出栈两个数据，根据符号计算相关结果，再把结果进栈

### 代码实现
```
#include <iostream>
#include <string>
#include <cstring>

using namespace std;

// 中缀变后缀
class Transform{

private:
    char *str;  // 栈数组
    int top;    // 栈顶
    int size;   // 栈能容纳的最大数据

public:
    // 表达式求值的构造函数
    Transform(int size) {
        this->size = size;
        str = new char[size];
        top = -1;
    }

    // 栈是否为空
    bool isEmpty() {
        return top == -1;
    }

    // 栈是否已经满了
    bool isFull() {
        return top == this->size-1;
    }

    // 入栈函数
    void Push(char ch) {
        if(!this->isFull()) {
            this->str[++top] = ch;
        }
    }

    // 获得栈顶元素
    char Top() {
        return this->str[top];
    }
    
    // 出栈元素
    void Pop() {
        this->top--;
    }

    // 栈外元素优先级
    int icp(char ch) {
        int result = 0;
        if(ch == '#') {
            result = 0;
        } else if(ch == '(') {
            result = 6;
        } else if(ch == '*' || ch == '/') {
            result = 4;
        } else if(ch == '+' || ch == '-') {
            result = 2;
        } else if (ch == ')') {
            result = 1;
        }

        return result;
    }

    // 栈内运算符优先级
    int isp(char ch) {
        int result = 0;
        if(ch == '#') {
            result = 0;
        } else if(ch == '(') {
            result = 1;
        } else if(ch == '*' || ch == '/') {
            result = 5;
        } else if (ch == '+' || ch == '-') {
            result = 3;
        } else if (ch == ')') {
            result = 6;
        }

        return result;
    }

    // 中缀表达式转为后缀表达式
    string Transform_exp(string experssion) {
        // 在中缀表达式和栈中追加"#"表示结束
        experssion.append("#");
        this->Push('#');

        int cnt = 0;
        char * tmp, *c;

        // 中缀表达式
        c = new char[experssion.length()];
        // 后缀表达式
        tmp = new char[experssion.length()]; 
        strcpy(c,experssion.c_str());

        for (int i = 0; i < experssion.length(); i++) {
            // 如果是数字，直接输出到后缀表达式中
            if(c[i] >= '0' && c[i] <= '9') {
                tmp[cnt++] = c[i];
            } else {
                // 和栈顶元素比较优先级
                char ch = this->Top();

                // 栈顶元素优先级比栈外元素优先级低, 则栈外元素入栈
                if(this->isp(ch) < this->icp(c[i])) {
                    this->Push(c[i]);
                } else if(this->isp(ch) > this->icp(c[i])) {
                    // 栈顶元素在栈内的优先级比栈外元素的优先级高
                    // 栈内元素出栈，并输出到后缀表达式中，循环变量-1
                    tmp[cnt++] = ch;
                    this->Pop();
                    i--;
                } else {
                    // 栈内栈外优先级相等
                    // 运行到#号或者是 栈内左括号 栈外右括号, 出栈即可
                    this->Pop();
                }
            }
        }

        // 返回中缀表达式的字符串
        string after = string(tmp,cnt);
        return after;
    }
};

class Sum{

private:
    int *sum;
    int top;
    int size;

public:
    // 表达式求值的构造函数
    Sum(int size) {
        this->size = size;
        sum = new int[size];
        this->top = -1;
    }

    // 栈是否为空
    bool IsEmpty() {
        return this->top == -1;
    }

    // 栈是否已经满
    bool IsFull() {
        return this->top == this->size - 1;
    }

    // 入栈函数
    void Push(int num) {
        if(!this->IsFull()) {
            this->sum[++top] = num;
        }
    }

    // 获得栈顶元素
    int Top() {
        return this->sum[top];
    }

    // 出栈函数
    void Pop() {
        this->top--;
    }

    // 后缀表达式求和
    int Sum_after(string after) {
        char *s;
        s = new char[after.length()];
        strcpy(s,after.c_str());

        for (int i = 0; i < after.length(); i++ ) {
            if(s[i] >= '0' && s[i] <= '9') {
                this->Push(s[i] - '0');
            } else{
                int b = this->Top();
                this->Pop();
                int a = this->Top();
                this->Pop();

                int result = 0;

                switch(s[i]){
                    case '+': result = a + b; break;
                    case '-': result = a - b; break;
                    case '*': result = a * b; break;
                    case '/': result = a / b; break;
                }

                this->Push(result);
            }
        }
        return this->Top();
    }
};

int main()
{
    string exp;
    cout << "Please enter the expression: " << endl;
    cin >> exp;
    
    // 中缀变后缀
    Transform transform(exp.length());
    string after = transform.Transform_exp(exp);
    cout << "the after exp is: " << endl;
    cout << after << endl;

    // 计算结果
    Sum sum(after.length());
    int result = sum.Sum_after(after);
    cout << "the result is: " << endl;
    cout << result << endl;

}
```
