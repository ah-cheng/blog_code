### 链表分离法
基本思想: 将相应位置上冲突的所有关键字存储在同一个单链表中, 构造以所有可能地址为头的单链表地址.

### 代码实现

```
#include <iostream>
#include <vector>

using namespace std;

// 建立一个链表
struct node{
    int data;
    struct node *next;
};

typedef struct node Node;

// 构建类
class HashTable{
public:
    // 构造函数
    HashTable(int length){
        this->length = length;
        nodes.reserve(this->length);
        this->Max_prime = this->GetMaxPrime();
        for(int i=0;i<this->length;i++) {
            this->nodes.push_back(this->Create_nodes(i));
        }
    };

    Node * Create_nodes(int num) {
        Node * node = new Node;
        node->data = num;
        node->next = NULL;

        return node;
    }

    bool isPrime(int num) {
        bool flag = true;

        if (num <= 1)
            return false;
        else if( num == 2)
            return false;
        else {
            for (int i = 2 ; i < num - 1 ; i++ ) {
                if ( num % i == 0 ) 
                    flag = false;
                    break;
            }
        }
        
        return flag;
    }

    int GetMaxPrime() {
        for(int i = this->length - 1; i >= 0; i--) {
            if ( this-> isPrime(i) )
                return i;
        }

        return 0;
    }

    // 哈希函数
    int hash (int num) {
        return num % this->Max_prime;
    }

    // 关联节点
    node * Attach (Node *node1, Node * node2 ) {
        node *cur = node1;
        while (1) {
            if ( cur->next == NULL) {
                cur->next = node2;
                break;
            }

            cur = cur-> next;
        }

        return node1;
    }

    // 插入表
    void Inset(int num) {
        // 计算节点所在地址
        int index = this->hash(num);

        // 建立一个节点
        Node *node = NULL;
        node = this->Create_nodes(num);

        // 把节点接到合适的地址位置上
        Node *temp = NULL;
        temp = this-> nodes[index];
        this->nodes[index] = this-> Attach(temp,node); 
    }

    void PrintList( Node * node) {
        Node * cur = node;
        while(cur) {
            cout << cur->data << " ";
            cur = cur->next;
        }
    }

    // 打印表
    void Print() {
        cout << "index\tvalue: " << endl;
        for(int i = 0; i< this->nodes.size(); i++) {
            if (this->nodes[i]->next != NULL) {
                cout << i << ":\t";
                this->PrintList(this->nodes[i]->next);
                cout << endl;
            } else {
                cout << i << ":\t " << endl;
            }
        }
    }

    void Create(int size, int *data) {
        for( int i = 0; i < size; i++) {
            this->Inset(data[i]);
        }
    }


    int Find( int num) {
        int index;

        // 找到num对应的地址
        index = this->hash(num);

        // 循环检测数据是否存在
        Node * cur = this-> nodes[index];

        while (cur->next) {
            if ( cur->next->data == num ) {
                return index;
            }
            cur = cur->next;
        }

        return -1;
    }

    int Delete(int num) {
        // 找到当前节点的index
        int index = this->hash(num);

        Node *cur = this->nodes[index];
        Node *tmp;
        int count = 0;
        while (cur->next != NULL) {
            if (cur->next->data == num) {
                tmp = cur->next;
                cur->next = tmp->next;
                delete tmp;
                count++;
            } else {
                cur = cur->next;
            }
        }

        return count;
    }


private:
    int length;                 // 表长
    vector<node *> nodes;       // 输入的数据
    int Max_prime;              // 最大素数，作为哈希函数
};

int main()
{
    int length;
    int size;
    int * data;

    cout << "please enter hash  table length: " << endl;
    cin >> length;
    cout << "Please enter the element size: " << endl;
    cin >> size;
    cout << "Please init the data: " << endl;

    data = new int[size];
    for (int i = 0; i < size; i++) {
        cin >> data[i];
    }

    HashTable hashtable(length);

    hashtable.Create(size,data);
    hashtable.Print();

    // 查找数据
    int num;
    cout << "Please enter the nember of find: " << endl;
    cin >> num;
    int index = hashtable.Find(num);
    if ( index == -1) {
        cout << " The number " << num << " not found " << endl;
    } else {
        cout << " The number " << num << " is in index: " << index << endl;
    }

    // 插入数据
    cout << "Please enter you want to inset num:" << endl;
    cin >> num;
    cout << "hashtable value before insert: " << endl;
    hashtable.Print();
    hashtable.Inset(num);
    hashtable.Print();
    

    // 删除数据
    cout << "Please enter you want to delete num: " << endl;
    cin >> num;
    int count = hashtable.Delete(num);
    cout << "delete " << count << " values. " << endl;
    hashtable.Print();

    return 0;
}
```
